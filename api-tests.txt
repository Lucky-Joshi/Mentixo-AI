1. AUTH API (public — no token needed)
----------------------------------------------------

-- Signup --
Method  : POST
URL     : http://localhost:5000/api/auth/signup
Headers : Content-Type: application/json
Body:
{
  "name": "Lucky",
  "email": "lucky@example.com",
  "password": "secret123"
}

Expected Response (201):
{
  "success": true,
  "token": "<jwt_token>",
  "user": { "id": "...", "name": "Lucky", "email": "lucky@example.com" }
}

-- Signup duplicate email (should return 409) --
Body:
{
  "name": "Lucky",
  "email": "lucky@example.com",
  "password": "secret123"
}

-- Signup missing fields (should return 400) --
Body:
{
  "email": "lucky@example.com"
}

-- Login --
Method  : POST
URL     : http://localhost:5000/api/auth/login
Headers : Content-Type: application/json
Body:
{
  "email": "lucky@example.com",
  "password": "secret123"
}

Expected Response (200):
{
  "success": true,
  "token": "<jwt_token>",
  "user": { "id": "...", "name": "Lucky", "email": "lucky@example.com" }
}

-- Login wrong password (should return 401) --
Body:
{
  "email": "lucky@example.com",
  "password": "wrongpassword"
}


----------------------------------------------------
NOTE: For all routes below, copy the token from
signup/login and add this header to every request:

  Authorization: Bearer <your_jwt_token>
----------------------------------------------------

====================================================
  CURL COMMANDS (Windows - use double quotes)
====================================================

Signup:
curl -X POST http://localhost:5000/api/auth/signup -H "Content-Type: application/json" -d "{\"name\":\"Lucky\",\"email\":\"lucky@example.com\",\"password\":\"secret123\"}"

Login:
curl -X POST http://localhost:5000/api/auth/login -H "Content-Type: application/json" -d "{\"email\":\"lucky@example.com\",\"password\":\"secret123\"}"

Chat (replace TOKEN):
curl -X POST http://localhost:5000/api/chat -H "Content-Type: application/json" -H "Authorization: Bearer TOKEN" -d "{\"prompt\":\"Explain recursion\"}"

Notes (replace TOKEN):
curl -X POST http://localhost:5000/api/notes -H "Content-Type: application/json" -H "Authorization: Bearer TOKEN" -d "{\"topic\":\"Operating System Scheduling\"}"

Quiz (replace TOKEN):
curl -X POST http://localhost:5000/api/quiz -H "Content-Type: application/json" -H "Authorization: Bearer TOKEN" -d "{\"topic\":\"DBMS\",\"difficulty\":\"medium\"}"
